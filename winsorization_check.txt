def analyze_dampening_overlap(Opex_data, exclude_set,
  percentile_dict, id_column_name):
      """
      Analyze overlap between exclusion dampening and
  percentile winsorization.
      """
      results = []

      for segment in Opex_data["TOPS"].unique():
          segment_data = Opex_data[Opex_data["TOPS"] ==
  segment].copy()
          quantile_value = percentile_dict[segment]

          # Create masks
          in_exclude_set =
  segment_data[["DATE_OF_FINANCIALS", id_column_name]].apply(
              tuple, axis=1
          ).isin(exclude_set)

          above_quantile = segment_data["delta_opex/rev"] >
  quantile_value
   
          # Categories
          only_excluded = in_exclude_set & ~above_quantile  
          only_winsorized = ~in_exclude_set & above_quantile
          both = in_exclude_set & above_quantile
          neither = ~in_exclude_set & ~above_quantile        

          results.append({
              'segment': segment,
              'n_total': len(segment_data),
              'n_only_excluded': only_excluded.sum(),        
              'n_only_winsorized': only_winsorized.sum(),    
              'n_OVERLAP': both.sum(),  # KEY: These get    
  different treatment
              'n_neither': neither.sum(),
              'pct_overlap': 100 * both.sum() /
  len(segment_data),
              'quantile_threshold': quantile_value,
              'mean_overlap_value': segment_data.loc[both,  
  "delta_opex/rev"].mean() if both.any() else np.nan,        
              'median_overlap_value': segment_data.loc[both,
  "delta_opex/rev"].median() if both.any() else np.nan,      
          })

      return pd.DataFrame(results)

  def compare_evt_parameters(data_A, data_B, segments,      
  threshold_percentile=0.95):
      """
      Compare EVT parameters between scenarios.
      """
      from scipy.stats import genpareto

      results = []

      for segment in segments:
          for scenario, data, label in [(data_A,
  'A_exclude'), (data_B, 'B_include')]:
              segment_data = data[data["TOPS"] ==
  segment]["delta_opex/rev"].dropna()

              threshold =
  segment_data.quantile(threshold_percentile)
              exceedances = segment_data[segment_data >      
  threshold] - threshold

              if len(exceedances) >= 30:
                  try:
                      xi, loc, sigma =
  genpareto.fit(exceedances, floc=0)

                      results.append({
                          'segment': segment,
                          'scenario': label,
                          'threshold': threshold,
                          'n_exceedances': len(exceedances),
                          'xi_shape': xi,
                          'sigma_scale': sigma,
                          'mean_excess': exceedances.mean(),
                      })
                  except:
                      pass

      df = pd.DataFrame(results)

      # Pivot for easier comparison
      pivot = df.pivot_table(
          index='segment',
          columns='scenario',
          values=['xi_shape', 'sigma_scale', 'n_exceedances',
   'threshold']
      )

      return df, pivot

  5. Business Logic Decision Framework

  The key question is: Should values in exclude_set be      
  winsorized?

  Arguments for EXCLUDING from winsorization (current code):
  - Avoids double-dampening: exclude_set already dampened at
  line 1959
  - Preserves the original dampening intent for flagged      
  outliers
  - The exclusion list is manually curated, so those values  
  have special treatment

  Arguments for INCLUDING in winsorization:
  - Ensures consistency: ALL values above quantile are      
  treated the same
  - The first dampening (multiply by dampner_val) may not be
  enough
  - Winsorization is a separate outlier control mechanism
